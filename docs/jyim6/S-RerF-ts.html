sample_ts <- function(n, d, p, method="simple-poisson") {
  # Samples "n" time series with "d" times points.
  # Method is the sample procedure to use where 
  # a sample X_i ~ F_1 if Y_i = 1 
  # and X_i ~ F_2  if Y_i = 0
  # where Y_i ~ Bernoulli(p). This way n*p samples 
  # from F_1 and n*(1-p) samples will be from F_2
  
  sample <- list()
  if (method == 'simple-poisson') {
    # Samples time series from either a simple stationary 
    # poisson process Poiss(0.3)+N(0,0.1) if Y=1 
    # or from a N(0,0.1) if Y=0

    sample$Y <- rbinom(n, 1, p)
    sample_x <- function(x) {
      Y <<- sample$Y
      if (Y[x] == 1) {
        return(rbinom(d, 1, 0.3) + rnorm(d, 0, 0.1))
      } else{
        return(rnorm(d,0,0.1))
      }
    }
    sample$X <- t(sapply(1:n, sample_x))
  }
  return(sample)
}
